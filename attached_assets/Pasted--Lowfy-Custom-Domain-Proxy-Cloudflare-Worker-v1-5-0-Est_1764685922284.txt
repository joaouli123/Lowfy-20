/**
 * Lowfy Custom Domain Proxy - Cloudflare Worker v1.5.0
 * 
 * Este Worker permite que usuários do Lowfy usem seus próprios domínios
 * para acessar páginas clonadas/presell.
 * 
 * CHANGELOG v1.5.0:
 * - Adicionado suporte a challenge de verificação HTTP do Cloudflare
 * - Suporte a .well-known/cf-custom-hostname-challenge
 * - Suporte a .well-known/acme-challenge
 * - Melhor tratamento de domínios apex (raiz)
 */

// Em produção, use a URL publicada do seu app
// Para desenvolvimento, troque para a URL do Replit
const LOWFY_ORIGIN = 'https://lowfy.com.br';
const LOWFY_PUBLIC = 'https://lowfy.com.br';

// Armazenar challenges de verificação (em produção, usar KV)
const VERIFICATION_CHALLENGES = {
  // Formato: 'hostname': { challenge_id: 'body' }
};

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);
    const hostname = url.hostname.toLowerCase();
    // NÃO normaliza www - cada subdomínio é único (www, loja, app são diferentes)
    const pathname = url.pathname;
    
    // Bypass para domínio principal do Lowfy
    if (hostname === 'lowfy.com.br' || hostname === 'www.lowfy.com.br') {
      return fetch(request);
    }
    
    // Se for o próprio domínio do Worker ou proxy, retornar status
    if (hostname.includes('workers.dev') || hostname === 'proxy.lowfy.com.br') {
      return new Response(JSON.stringify({
        status: 'ok',
        service: 'Lowfy Custom Domain Proxy',
        version: '1.5.0',
        instructions: 'Configure seu domínio com CNAME apontando para proxy.lowfy.com.br'
      }), {
        headers: { 'Content-Type': 'application/json' }
      });
    }
    
    // IMPORTANTE: Responder challenges de verificação do Cloudflare
    if (pathname.startsWith('/.well-known/cf-custom-hostname-challenge/')) {
      const challengeId = pathname.split('/').pop();
      // Tentar buscar o challenge do backend
      try {
        const challengeResponse = await fetch(`${LOWFY_ORIGIN}/api/cloudflare-challenge/${hostname}/${challengeId}`);
        if (challengeResponse.ok) {
          const data = await challengeResponse.json();
          if (data.body) {
            return new Response(data.body, {
              status: 200,
              headers: { 'Content-Type': 'text/plain' }
            });
          }
        }
      } catch (e) {
        console.error('Challenge lookup error:', e);
      }
      // Retornar 404 se não encontrar
      return new Response('Challenge not found', { status: 404 });
    }
    
    // Responder challenges ACME para SSL
    if (pathname.startsWith('/.well-known/acme-challenge/')) {
      const token = pathname.split('/').pop();
      try {
        const challengeResponse = await fetch(`${LOWFY_ORIGIN}/api/acme-challenge/${hostname}/${token}`);
        if (challengeResponse.ok) {
          const data = await challengeResponse.json();
          if (data.body) {
            return new Response(data.body, {
              status: 200,
              headers: { 'Content-Type': 'text/plain' }
            });
          }
        }
      } catch (e) {
        console.error('ACME challenge lookup error:', e);
      }
      return new Response('ACME challenge not found', { status: 404 });
    }
    
    try {
      // Consultar API do Lowfy para descobrir qual página servir
      const lookupResponse = await fetch(`${LOWFY_ORIGIN}/api/domain-lookup/${encodeURIComponent(hostname)}`, {
        headers: {
          'Host': 'lowfy.com.br',
          'User-Agent': 'Lowfy-Proxy/1.5.0'
        }
      });
      
      if (!lookupResponse.ok) {
        console.error('Lookup failed:', lookupResponse.status);
        return new Response('Erro ao consultar domínio', { status: 502 });
      }
      
      const lookupData = await lookupResponse.json();
      
      if (!lookupData.found) {
        return new Response(`
          <!DOCTYPE html>
          <html lang="pt-BR">
          <head>
            <meta charset="UTF-8">
            <meta name="viewport" content="width=device-width, initial-scale=1.0">
            <title>Domínio não configurado</title>
            <style>
              body { font-family: system-ui, sans-serif; display: flex; align-items: center; justify-content: center; min-height: 100vh; margin: 0; background: #f5f5f5; }
              .container { text-align: center; padding: 40px; background: white; border-radius: 12px; box-shadow: 0 4px 20px rgba(0,0,0,0.1); max-width: 500px; }
              h1 { color: #29654F; margin-bottom: 16px; }
              p { color: #666; line-height: 1.6; }
              a { color: #29654F; }
            </style>
          </head>
          <body>
            <div class="container">
              <h1>Domínio não configurado</h1>
              <p>O domínio <strong>${hostname}</strong> não está vinculado a nenhuma página no Lowfy.</p>
              <p>Se você é o dono deste domínio, acesse sua conta no <a href="${LOWFY_PUBLIC}">Lowfy</a> e configure o domínio na sua página.</p>
            </div>
          </body>
          </html>
        `, {
          status: 404,
          headers: { 'Content-Type': 'text/html; charset=utf-8' }
        });
      }
      
      // Construir URL da página no Lowfy
      const targetUrl = `${LOWFY_ORIGIN}${lookupData.path}${url.search}`;
      
      // Fazer requisição para o Lowfy com headers especiais
      const proxyRequest = new Request(targetUrl, {
        method: request.method,
        headers: {
          'Accept': request.headers.get('Accept') || '*/*',
          'Accept-Language': request.headers.get('Accept-Language') || 'pt-BR',
          'User-Agent': request.headers.get('User-Agent') || 'Lowfy-Proxy/1.5.0',
          'X-Original-Host': hostname,
          'X-Forwarded-Host': hostname,
          'X-Forwarded-Proto': 'https',
          'Host': 'lowfy.com.br'
        },
        body: request.method !== 'GET' && request.method !== 'HEAD' ? request.body : null,
        redirect: 'manual'
      });
      
      const response = await fetch(proxyRequest);
      
      // Copiar resposta e ajustar headers
      const responseHeaders = new Headers(response.headers);
      responseHeaders.set('X-Served-By', 'Lowfy-Proxy');
      responseHeaders.set('X-Proxy-Version', '1.5.0');
      responseHeaders.delete('x-frame-options');
      
      // Processar a resposta
      let body = response.body;
      const contentType = response.headers.get('content-type') || '';
      
      // Se for HTML, ajustar URLs
      if (contentType.includes('text/html')) {
        const html = await response.text();
        body = html
          .replace(/https:\/\/lowfy\.com\.br\/pages\//g, '/')
          .replace(/https:\/\/lowfy\.com\.br\/presell\//g, '/')
          .replace(/href="https:\/\/lowfy\.com\.br\/(assets|uploads|static)/g, `href="${LOWFY_PUBLIC}/$1`)
          .replace(/src="https:\/\/lowfy\.com\.br\/(assets|uploads|static)/g, `src="${LOWFY_PUBLIC}/$1`);
      }
      
      // Ajustar Location header em caso de redirect
      if (responseHeaders.has('location')) {
        const location = responseHeaders.get('location');
        if (location) {
          const newLocation = location
            .replace(/https:\/\/lowfy\.com\.br\/pages\/[^\/\?]+/g, `https://${hostname}`)
            .replace(/https:\/\/lowfy\.com\.br\/presell\/[^\/\?]+/g, `https://${hostname}`);
          responseHeaders.set('location', newLocation);
        }
      }
      
      return new Response(body, {
        status: response.status,
        statusText: response.statusText,
        headers: responseHeaders
      });
      
    } catch (error) {
      console.error('Proxy error:', error);
      return new Response(`
        <!DOCTYPE html>
        <html lang="pt-BR">
        <head>
          <meta charset="UTF-8">
          <title>Erro temporário</title>
          <style>
            body { font-family: system-ui, sans-serif; display: flex; align-items: center; justify-content: center; min-height: 100vh; margin: 0; background: #f5f5f5; }
            .container { text-align: center; padding: 40px; background: white; border-radius: 12px; box-shadow: 0 4px 20px rgba(0,0,0,0.1); }
            h1 { color: #e53e3e; }
          </style>
        </head>
        <body>
          <div class="container">
            <h1>Erro temporário</h1>
            <p>Não foi possível carregar a página. Tente novamente em alguns segundos.</p>
            <p style="color: #999; font-size: 12px;">Proxy v1.5.0</p>
          </div>
        </body>
        </html>
      `, {
        status: 502,
        headers: { 'Content-Type': 'text/html; charset=utf-8' }
      });
    }
  }
};
